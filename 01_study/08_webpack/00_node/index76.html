EJS scripts 문법 - 반복문
EJS에서는 자바스크립트 코드를 <% %> 안에 씁니다.
따라서 반복문도 이 안에 쓰면 됩니다.
Pug 처럼 따로 문법이 있는 것은 아니고, for나 while 같은 반복문을 사용하면 됩니다. 
                    
                        <ul>
                            <% var fruits = ['사과', '배', '오랜지', '바나나', '복숭아'] %>
                            <% for (var i = 0; i < fruits.length; i++) { %>
                            <li><%= fruits[i] %></li>
                            <% } %>
                        </ul>
                    
                
                    
                        <ul>
                            <% var fruits = ['사과', '배', '오랜지', '바나나', '복숭아'] %>
                            <% for (var i = 0; i < fruits.length; i++) { %>
                            <li><%= (i + 1) + '번째' + fruits[i] %></li>
                            <% } %>
                        </ul>
